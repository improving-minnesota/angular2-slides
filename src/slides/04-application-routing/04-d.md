## `@Output`

```typescript
import { Component, EventEmitter, Output } from '@angular/core';

@Component({
  selector: 'app-dog-list',
  templateUrl: './dog-list.component.html'
})
export class DogListComponent {
  @Output()
  public onAdd: EventEmitter<IDog>: new EventEmitter<IDog>();

  @Input()
  public dogs: IDog[] = [];

  addDog(dog: IDog) {
    this.dogs.push(dog);
    this.onAdd.emit(dog);
  }
}
```
